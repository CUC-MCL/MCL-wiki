    #二叉树创建和先序遍历，有可能会用到类
class Node(object):
    def __init__(self,value='',lchild=None,rchild=None):
        self.value=value
        self.lchild=lchild
        self.rchild=rchild
class BiTree(object):
    def __init__(self,nodes=''):
        self.root=Node()
        for node in nodes:
            self.add(node)
    def add(self,value):
        if self.root.value=='':
            self.root.value=value
        else:
            queue=[self.root]
            while queue!=[] and self.root:
                if queue[0].lchild==None:
                    queue[0].lchild=Node(value)
                    return
                else:
                    queue.append(queue[0].lchild)
                if queue[0].rchild==None:
                    queue[0].rchild=Node(value)
                    return
                else:
                    queue.append(queue[0].rchild)
                queue.pop(0)
    def preOrderTraverse(self,root):#递归几乎没区别
        if root == None:
            return
        print(root.value)
        self.preOrderTraverse(root.lchild)
        self.preOrderTraverse(root.rchild)
